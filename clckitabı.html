-- Supabase SQL Tabloları
CREATE TABLE messages (
    id BIGSERIAL PRIMARY KEY,
    content TEXT NOT NULL,
    username TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE active_users (
    username TEXT PRIMARY KEY,
    last_seen TIMESTAMPTZ DEFAULT NOW()
);

-- 10 saniye sonra aktif kullanıcıları temizle
CREATE OR REPLACE FUNCTION cleanup_old_users()
RETURNS TRIGGER AS $$
BEGIN
    DELETE FROM active_users 
    WHERE last_seen < NOW() - INTERVAL '10 seconds';
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger'ı aktif et
CREATE TRIGGER trigger_cleanup_users
    AFTER INSERT OR UPDATE ON active_users
    EXECUTE FUNCTION cleanup_old_users();

2.2 Supabase HTTP Servisi

// SupabaseHttpService.cs
using System;
using System.Net.Http;
using System.Text;
using System.Threading.Tasks;
using Newtonsoft.Json;
using Newtonsoft.Json.Linq;

public class SupabaseHttpService
{
    private readonly HttpClient _httpClient;
    private readonly string _supabaseUrl = "https://zsspdxeovynnzrjalvvy.supabase.co";
    private readonly string _supabaseKey = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...";
    
    public SupabaseHttpService()
    {
        _httpClient = new HttpClient();
        _httpClient.DefaultRequestHeaders.Add("apikey", _supabaseKey);
        _httpClient.DefaultRequestHeaders.Add("Authorization", $"Bearer {_supabaseKey}");
    }
    
    public async Task GetAsync(string endpoint)
    {
        var response = await _httpClient.GetAsync($"{_supabaseUrl}/rest/v1/{endpoint}");
        var content = await response.Content.ReadAsStringAsync();
        return JArray.Parse(content);
    }
    
    public async Task PostWithReturnAsync(string endpoint, object data)
    {
        var json = JsonConvert.SerializeObject(data);
        var content = new StringContent(json, Encoding.UTF8, "application/json");
        var request = new HttpRequestMessage(HttpMethod.Post, $"{_supabaseUrl}/rest/v1/{endpoint}")
        {
            Content = content
        };
        request.Headers.Add("Prefer", "return=representation");
        
        var response = await _httpClient.SendAsync(request);
        var responseContent = await response.Content.ReadAsStringAsync();
        return JArray.Parse(responseContent);
    }
}

2.3 ChatService (Ana İş Mantığı)

// ChatService.cs
public class ChatService
{
    private readonly SupabaseHttpService _httpService = new();
    private readonly List _badWords = new()
    {
        "küfür", "kufur", "piç", "pic", "amk", "aq"
    };
    
    private string _currentUser = "";
    private DateTime _joinTime;
    
    public ObservableCollection Messages { get; } = new();
    public ObservableCollection ActiveUsers { get; } = new();
    
    public async Task LoginAsync(string username)
    {
        // 10 saniyeden eski kullanıcıları temizle
        var tenSecondsAgo = DateTime.UtcNow.AddSeconds(-10);
        await _httpService.DeleteAsync($"active_users?last_seen=lt.{tenSecondsAgo:yyyy-MM-dd HH:mm:ss}");
        
        // Kullanıcı adı kullanımda mı?
        var existing = await _httpService.GetAsync($"active_users?username=eq.{username}");
        if (existing.HasValues) return false;
        
        _currentUser = username;
        _joinTime = DateTime.UtcNow;
        
        // Giriş mesajı gönder
        await _httpService.PostAsync("messages", new
        {
            content = $"🚪 {username} katıldı!",
            username = "SİSTEM",
            created_at = DateTime.UtcNow
        });
        
        return true;
    }
}

4.1 Polling ile Mesaj Takibi

public void StartPolling()
{
    Task.Run(async () =>
    {
        while (_isPolling)
        {
            await Task.Delay(2000); // 2 saniyede bir kontrol
            
            // Yeni mesajları yükle
            await LoadMessagesAsync();
            
            // Aktif kullanıcıları güncelle
            await LoadActiveUsersAsync();
            
            // LastSeen'i güncelle
            await UpdateLastSeenAsync();
        }
    });
}

public async Task LoadMessagesAsync()
{
    var response = await _httpService.GetAsync(
        $"messages?select=*&created_at=gt.{_joinTime:yyyy-MM-dd HH:mm:ss}&order=created_at.asc");
    
    MainThread.BeginInvokeOnMainThread(() =>
    {
        Messages.Clear();
        foreach (var item in response)
        {
            Messages.Add(new Message
            {
                Id = item["id"]?.Value<int>() ?? 0,
                Content = item["content"]?.Value<string>() ?? "",
                Username = item["username"]?.Value<string>() ?? "",
                CreatedAt = DateTime.Parse(item["created_at"]?.Value<string>() ?? DateTime.UtcNow.ToString()),
                IsOwnMessage = item["username"]?.Value<string>() == _currentUser
            });
        }
    });
}

4.2 Küfür Filtresi

private readonly List _badWords = new()
{
    "küfür", "kufur", "piç", "pic", "amk", "aq", "sik", "siktir",
    "ananı", "babanı", "gerizekalı", "salak", "aptal", "ibne",
    "orospu", "çocuğu", "cocugu", "yarrak", "amcık", "amcik"
};

public bool ContainsBadWord(string text)
{
    var lowerText = text.ToLower();
    return _badWords.Any(w => lowerText.Contains(w.ToLower()));
}

4.3 /stats Komutu

private async Task ShowStatsAsync()
{
    var response = await _httpService.GetAsync(
        $"messages?select=username&created_at=gt.{_joinTime:yyyy-MM-dd HH:mm:ss}");
    
    var messages = response.Select(x => x["username"]?.Value<string>() ?? "").ToList();
    
    var total = messages.Count;
    var myMessages = messages.Count(x => x == _currentUser);
    
    var topUsers = messages
        .Where(x => x != "SİSTEM")
        .GroupBy(x => x)
        .Select(g => new { User = g.Key, Count = g.Count() })
        .OrderByDescending(x => x.Count)
        .Take(3)
        .ToList();
    
    var stats = $"📊 İSTATİSTİKLER\nToplam: {total}\nSenin: {myMessages}\n\n🏆 En Çok:\n";
    for (int i = 0; i < topUsers.Count; i++)
        stats += $"{i+1}. {topUsers[i].User}: {topUsers[i].Count} mesaj\n";
    
    await _httpService.PostAsync("messages", new
    {
        content = stats,
        username = "SİSTEM",
        created_at = DateTime.UtcNow
    });
}

6.1 Yaygın Hatalar ve Çözümleri

Hata: "Cannot add extension with name 'kotlin'"

# Çözüm: build.gradle.kts dosyasını temizleyip yeniden oluşturun
cd android
rm build.gradle.kts
# Yeni build.gradle.kts oluşturun

Hata: "No space left on device"

# bin ve obj klasörlerini temizle
rm -r -force bin
rm -r -force obj

Hata: "AAPT2 error" (Android derleme hatası)

# Çözüm: Android SDK'yı güncelle ve önbelleği temizle
